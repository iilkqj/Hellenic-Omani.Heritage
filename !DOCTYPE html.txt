    function renderSharedWords() {
        const words = [ {ar:"صندوق - σαντούκι", en:"Sanduq - sandoúki", el:"σαντούκι"}, {ar:"قلب - καρδιά", en:"Heart - kardia", el:"καρδιά"}, {ar:"قيراط - καράτι", en:"Carat - karati", el:"καράτι"}, {ar:"قافلة - καραβάνι", en:"Caravan - karavani", el:"καραβάνι"} ];

    window.showModal = function(siteId) {
        const s = sitesData[siteId]; if(!s) return;
        document.getElementById('modalTitle').innerHTML = s[`name_${currentLang}`];
        document.getElementById('modalInfo').innerHTML = `<strong>ℹ️ ${currentLang==='ar'?'المعلومات':(currentLang==='en'?'Info':'Πληροφορίες')}:</strong> ${s[`info_${currentLang}`]}`;
        document.getElementById('modalVisit').innerHTML = `<strong>📅 ${currentLang==='ar'?'أفضل وقت للزيارة':(currentLang==='en'?'Best visit':'Καλύτερη επίσκεψη')}:</strong> ${s[`visit_${currentLang}`]}`;
        const googleMapsUrl = `https://www.google.com/maps?q=${s.lat},${s.lng}`;
        document.getElementById('modalMapLink').href = googleMapsUrl;
        const galleryDiv = document.getElementById('modalGalleryThumbs'); galleryDiv.innerHTML = '';
        if(s.images && s.images.length) { s.images.forEach(img => { const thumb = document.createElement('img'); thumb.src = img; thumb.classList.add('thumb-img'); thumb.onclick = () => document.getElementById('mainModalImg').src = img; galleryDiv.appendChild(thumb); }); document.getElementById('mainModalImg').src = s.images[0]; }
        document.getElementById('siteModal').style.display = 'block';
    }
    function showPage(pageId) {
        document.querySelectorAll('.page').forEach(p => p.classList.remove('active-page'));
        document.getElementById(`${pageId}-page`).classList.add('active-page');
        document.querySelectorAll('.nav-links a').forEach(a => a.classList.remove('active-page'));
        document.querySelector(`.nav-links a[data-page="${pageId}"]`).classList.add('active-page');
        if(pageId === 'oman') renderOman(); if(pageId === 'greece') renderGreece(); if(pageId === 'home') { renderOman(); renderGreece(); renderSharedWords(); }
    }
    document.querySelectorAll('.nav-links a').forEach(link => { link.addEventListener('click', (e) => { e.preventDefault(); showPage(link.getAttribute('data-page')); }); });
    document.querySelectorAll('.country-card').forEach(card => { card.addEventListener('click', () => showPage(card.getAttribute('data-country'))); });
    document.querySelectorAll('.lang-btn').forEach(btn => { btn.addEventListener('click', () => { currentLang = btn.getAttribute('data-lang'); document.querySelectorAll('.lang-btn').forEach(b => b.classList.remove('active')); btn.classList.add('active'); updateUILanguage(); if(document.getElementById('oman-page').classList.contains('active-page')) renderOman(); if(document.getElementById('greece-page').classList.contains('active-page')) renderGreece(); renderSharedWords(); }); });
    document.querySelector('.close-modal').onclick = () => document.getElementById('siteModal').style.display = 'none';
    window.onclick = e => { if(e.target === document.getElementById('siteModal')) document.getElementById('siteModal').style.display = 'none'; };
    renderOman(); renderGreece(); renderSharedWords();
